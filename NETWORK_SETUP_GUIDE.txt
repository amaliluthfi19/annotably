// - data/network/NetworkResult.kt - Sealed class for network responses
// - data/repository/BaseRepository.kt - Base repository with safe API call handling
// - data/repository/ExampleRepository.kt - Example repository with constructor injection
// - utils/CoroutineUtils.kt - Coroutine utility functions
